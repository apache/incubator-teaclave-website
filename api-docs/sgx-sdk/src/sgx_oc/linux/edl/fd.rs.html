        buf: *mut c_void,

        buf: *const c_void,
        count: size_t,
    ) -> sgx_status_t;

        offset: off64_t,
    ) -> sgx_status_t;

        count: size_t,

        off_out: *mut off64_t,
        len: size_t,
        flags: c_uint,

        result: *mut ssize_t,

        off_out: *mut off64_t,
        len: size_t,
        flags: c_uint,
    ) -> sgx_status_t;
    pub fn u_fcntl_arg0_ocall(
        result: *mut c_int,
        errno: *mut c_int,
        fd: c_int,
        cmd: c_int,
    ) -> sgx_status_t;
    pub fn u_fcntl_arg1_ocall(
        result: *mut c_int,
        errno: *mut c_int,
        fd: c_int,
        cmd: c_int,
        arg: c_int,
    ) -> sgx_status_t;
    pub fn u_ioctl_arg0_ocall(

        fd: c_int,

    pub fn u_ioctl_arg1_ocall(
        result: *mut c_int,

    ) -> sgx_status_t;

        initval: c_uint,
        flags: c_int,
    ) -> sgx_status_t;

}
